$("#checkBox").click(function(){
		if($(this).html()==""){
			$(this).html("?")
		}else{
			$(this).html("")
		}
	})

	width:40px;
	height:40px;
	font-size:40px;
	color:#000;

	text-align:center;
	float:left;
	font-weight:bold;
	font-family:"Arial Black", Gadget, sans-serif;
	line-height:35px;
	cursor:pointer;
}
